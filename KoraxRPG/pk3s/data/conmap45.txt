conitem k24501 24501
{
	name "Commander Pablo";
	pic con24501;

	var MyState 1 integer 0;

	hello
	{
		say "Hail, hero!" none;
	}
	
	choice none "What can you tell me about the south?"
	{
		say Player "What can you tell me about the lands beyond the border?" none;
		say "To reach The Canyon, you have to traverse the Chasm. The plank bridge is in a bad shape so watch your step." none;
		say "From The Canyon, you can travel to the hills and mountains to the east and west, or continue south to The Rift. I have no idea what lies beyond The Rift." none;
		Player.SetMapTrackdb( 1 );
		Player.SetMapTrackdi( 1 );
		say "Anything else?" none;
	}

	choice none "Got anything to help me?"
	if (MyState is 0)
	{
		say Player "Do you have anything that could help me?" none;
		say "Lernal Cave to the east is supposed to hold a rare weapon." none;
		Player.StartQuestce( 1 );
		Player.SetMapTrackdh( 1 );
		Player.GiveConstitution( 2 );
		executems 45 2;
		set MyState 1;
	}


	choice none "My armor is in tatters."
	{
		say Player "My armor is in tatters." none;
		Player.GiveInventory( MeshArmor , 1 );
		say "Here some scraps salvaged off a corpse. Anything else?" none;
	}
	
	choice none "I'm ready to go."
	{
		say "May your axe bathe in blood." none;
		Player.SetMapTrackdm( 1 );
		end;
	}
}
