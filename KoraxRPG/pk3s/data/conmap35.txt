		say Player "I am back from Gnosis and bring the artifact for you." none;

		if (Player.HasInventory( WeaponClericMaceCommon ))
		{

		Player.GiveInventory( Coin, 250 );
		executems 35 5;
		terminate;
		}
		else if (Player.HasInventory( WeaponFighterSpikedGauntlets ))
		{
		say "Wonderful! Do you feel your mind expanding?" none;
		Player.GiveLevels( 5 );
		say "To care for your physical self as well, take this bag of coins." none;
		Player.GiveInventory( Coin, 250 );
		terminate;
		}

	}

	choice none "I'll be going now."
	{

		say "The Light be with you." none;
		end;
	}
