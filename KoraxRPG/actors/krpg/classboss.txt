//SETin warlord
actor KRPGFighterBoss : IntelligentActor replaces FighterBoss
{
	Game Hexen
	Health 800
	Radius 16
	Height 64
	Speed 25
	PainChance 50
	Experience 10000
	Monster
	+FloorClip
	+Telestomp
	+DontMorph
	DropItem "SilverBar", 40, 1
	SeeSound "SETinSight"
	ActiveSound "SETinActive"
	AttackSound "SETinAttack"
	PainSound "SETinPain"
	DeathSound "SETinDeath"
	HowlSound "PuppyBeat"
	Obituary "$ob_SETin"
	
	+CanJump
	+CanLeap
	+CanBackpedal
	+CanSidemove
	+CanDodge
	+FastMove
	+CanBerserk

	action native A_FighterAttack();

	states
	{
	Spawn:
		SETN A 0 A_JumpIf(Health < default_health / 2, "SpawnDamaged")
		SETN A 3 A_ClassBossHealth
		SETN AA 10 A_LookAI
		Loop
	SpawnDamaged:
		SETN S 0 A_JumpIf(Health > default_health / 2, "Spawn")
		SETN SS 10 A_LookAI
		Loop
	See:
		SETN A 0 A_JumpIf(Health < default_health / 2, "SeeDamaged")
		SETN ABCD 5 A_ChaseAI
		Loop
	SeeDamaged:
		SETN S 0 A_JumpIf(Health > default_health / 2, "See")
		SETN STUV 5 A_ChaseAI
		Loop
	Pain:
		SETN H 0 A_JumpIf(Health < default_health / 2, "PainDamaged")
		SETN H 7 A_KRPGPain
		Goto See
	PainDamaged:
		SETN Z 0 A_JumpIf(Health > default_health / 2, "Pain")
		SETN Z 7 A_KRPGPain
		Goto SeeDamaged
	Melee:
		SETN E 0 A_JumpIf(Health < default_health / 2, "MeleeDamaged")
		SETN EF 6 A_FaceTarget
		SETN G 8 A_CustomMeleeAttack(random(1, 8) * ((default_health - 175) / 175 + 2))
		Goto See
	MeleeDamaged:
		SETN W 0 A_JumpIf(Health > default_health / 2, "Melee")
		SETN WX 6 A_FaceTarget
		SETN Y 8 A_CustomMeleeAttack(random(1, 8) * ((default_health - 175) / 175 + 2))
		Goto SeeDamaged
	Missile:
		SETN E 0 A_JumpIf(Health < default_health / 2, "MissileDamaged")
		SETN EF 6 A_FaceTarget
		SETN F 8 A_FighterAttack
		Goto See
//		Goto Super::See
	MissileDamaged:
		SETN W 0 A_JumpIf(Health > default_health / 2, "Missile")
		SETN WX 6 A_FaceTarget
		SETN F 8 A_FighterAttack
		Goto See
//		Goto Super::See
	Death:
		SETN IJ 4
		SETN K 4 A_Scream
		SETN L 4 A_NoBlocking
		SETN M 4 A_QueueCorpse
		SETN NOP 4
		SETN Q -1
		Stop
	XDeath:
		SETB A 4
		SETB B 4 A_NoBlocking
		SETB C 4 A_SpawnItemEx(SETinMace, 0, 0, 34, random(-128, 127) * 0.03125,
			random(-128, 127) * 0.03125, 10 + random(0, 255) * 0.015625, 0,
			SXF_ABSOLUTEMOMENTUM)
		SETB D 4 A_Scream
		SETB E 4 A_QueueCorpse
		SETB FGHIJK 4
		SETB L -1
		Stop
	Ice:
		SETN R 5 A_FreezeDeath
		SETN R 1 A_FreezeDeathChunks
		Wait
	Burn:
		SETB A 4
		SETB B 4 A_NoBlocking
		SETB C 4 A_SpawnItemEx(SETinMace, 0, 0, 34, random(-128, 127) * 0.03125,
			random(-128, 127) * 0.03125, 10 + random(0, 255) * 0.015625, 0,
			SXF_ABSOLUTEMOMENTUM)
		SETB D 4 A_Scream
		SETB E 4 A_QueueCorpse
		SETB FGHIJK 4
		SETB L -1
		Wait
	BurnGone:
		ACLO E 8
		Stop
//	Burn:
//		FDTH A 5 Bright A_PlaySound("PlayerFighterBurnDeath")
//		FDTH B 4 Bright
//		FDTH G 5 Bright
//		FDTH H 4 Bright A_Scream
//		FDTH I 5 Bright
//		FDTH J 4 Bright
//		FDTH K 5 Bright
//		FDTH L 4 Bright
//		FDTH M 5 Bright
//		FDTH N 4 Bright
//		FDTH O 5 Bright
//		FDTH P 4 Bright
//		FDTH Q 5 Bright
//		FDTH R 4 Bright
//		FDTH S 5 Bright A_NoBlocking
//		FDTH T 4 Bright
//		FDTH U 5 Bright
//		FDTH V 4 Bright
//		ACLO E 35 A_CheckPlayerDone
//		Wait
//	BurnGone:
//		ACLO E 8
//		Stop
	}
}
