actor KArenaHereticImp : HereticImp 15009
{
	+Telestomp
	Obituary "$ob_gargoyle"
	HitObituary "$ob_gargoylehit"
	states
	{
	Crash:
		IMPX I 7 A_ImpExplode
		IMPX J 7 A_Scream
		IMPX K 7 A_QueueCorpse
		IMPX L -1
		Stop
	XCrash:
		IMPX X 7
		IMPX Y 7 A_QueueCorpse
		IMPX Z -1
		Stop
	Ice:
		IMPX "[" 5 A_FreezeDeath
		IMPX "[" 1 A_FreezeDeathChunks
		Wait
	}
}
